def text_to_binary(text):

    """–ö–æ–Ω–≤–µ—Ä—Ç—É—î –±—ñ–Ω–∞—Ä–Ω–∏–π —Ä—è–¥–æ–∫ –≤ —Ç–µ–∫—Å—Ç"""

def calculate_psnr(img1, img2):
    """–û–±—á–∏—Å–ª—é—î PSNR –º—ñ–∂ –¥–≤–æ–º–∞ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è–º–∏"""
    mse = np.mean((img1.astype(float) - img2.astype(float)) ** 2)
    if mse == 0:
        return float('inf')
    max_pixel = 255.0
    psnr = 20 * np.log10(max_pixel / np.sqrt(mse))
    return psnr

def dct2d_block(block):
    """
    –ó–∞—Å—Ç–æ—Å–æ–≤—É—î 2D DCT –¥–æ –±–ª–æ–∫—É 8x8

    Parameters:
    -----------

    Returns:
    --------
    dct_block : numpy array - DCT –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏
    """
    return dct(dct(block.T, norm='ortho').T, norm='ortho')

def idct2d_block(block):
    """
    –ó–∞—Å—Ç–æ—Å–æ–≤—É—î inverse 2D DCT –¥–æ –±–ª–æ–∫—É 8x8

    Parameters:
    -----------

def apply_dct_to_image(image):
    """
    –ó–∞—Å—Ç–æ—Å–æ–≤—É—î DCT –¥–æ –≤—Å—å–æ–≥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –±–ª–æ–∫–∞–º–∏ 8x8

    Parameters:
    -----------
    image : numpy array - –≤—Ö—ñ–¥–Ω–µ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è (grayscale)

    Returns:
    --------
    dct_image : numpy array - DCT –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏ –≤—Å—å–æ–≥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    """
    height, width = image.shape
    dct_image = np.zeros_like(image, dtype=np.float32)

    # –û–±—Ä–æ–±–∫–∞ –∫–æ–∂–Ω–æ–≥–æ –±–ª–æ–∫—É 8x8
    for i in range(0, height - 7, 8):
        for j in range(0, width - 7, 8):
            # –í–∏—Ç—è–≥—É—î–º–æ –±–ª–æ–∫
            block = image[i:i+8, j:j+8]

            # –ó—Å—É–≤ –Ω–∞ 128 (—Ü–µ–Ω—Ç—Ä—É–≤–∞–Ω–Ω—è)
            block = block.astype(np.float32) - 128

            # –ó–∞—Å—Ç–æ—Å–æ–≤—É—î–º–æ DCT
            dct_block = dct2d_block(block)

            # –ó–±–µ—Ä—ñ–≥–∞—î–º–æ —Ä–µ–∑—É–ª—å—Ç–∞—Ç
            dct_image[i:i+8, j:j+8] = dct_block

    return dct_image

def apply_idct_to_image(dct_image, original_shape):
    """
    –ó–∞—Å—Ç–æ—Å–æ–≤—É—î inverse DCT –¥–æ –≤—Å—å–æ–≥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è

    Parameters:
    -----------
    dct_image : numpy array - DCT –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏
    original_shape : tuple - —Ä–æ–∑–º—ñ—Ä –≤–∏—Ö—ñ–¥–Ω–æ–≥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è

    Returns:
    --------
    reconstructed : numpy array - –≤—ñ–¥–Ω–æ–≤–ª–µ–Ω–µ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    """
    height, width = original_shape
    reconstructed = np.zeros((height, width), dtype=np.float32)

    for i in range(0, height - 7, 8):
        for j in range(0, width - 7, 8):
            # –í–∏—Ç—è–≥—É—î–º–æ DCT –±–ª–æ–∫
            dct_block = dct_image[i:i+8, j:j+8]

            # Inverse DCT
            spatial_block = idct2d_block(dct_block)

            # –ó—Å—É–≤ –Ω–∞–∑–∞–¥ –Ω–∞ +128
            spatial_block = spatial_block + 128

            reconstructed[i:i+8, j:j+8] = spatial_block

    return reconstructed.astype(np.uint8)

    """
    –ì–µ–Ω–µ—Ä—É—î —Ç–∞–±–ª–∏—Ü—é –∫–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è –Ω–∞ –æ—Å–Ω–æ–≤—ñ —è–∫–æ—Å—Ç—ñ JPEG

    """
    # –ë–∞–∑–æ–≤–∞ —Ç–∞–±–ª–∏—Ü—è –∫–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è –¥–ª—è —è—Å–∫—Ä–∞–≤–æ—Å—Ç—ñ (JPEG standard)

        [16, 11, 10, 16, 24, 40, 51, 61],
        [12, 12, 14, 19, 26, 58, 60, 55],
        [14, 13, 16, 24, 40, 57, 69, 56],
        [14, 17, 22, 29, 51, 87, 80, 62],
        [18, 22, 37, 56, 68, 109, 103, 77],
        [24, 35, 55, 64, 81, 104, 113, 92],
        [49, 64, 78, 87, 103, 121, 120, 101],

    ], dtype=np.float32)

        scale = 5000 / quality
    else:

def jsteg_embed(image, message, quality=75):
    """
    –í–±—É–¥–æ–≤—É—î –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –º–µ—Ç–æ–¥–æ–º JSteg

    Parameters:
    -----------
    image : numpy array - –≤—Ö—ñ–¥–Ω–µ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è (grayscale)
    message : str - –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –¥–ª—è –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è
    quality : int - —è–∫—ñ—Å—Ç—å JPEG

    Returns:
    --------
    stego_image : numpy array - —Å—Ç–µ–≥–æ-–∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    capacity : int - –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–∞ —î–º–Ω—ñ—Å—Ç—å –≤ –±—ñ—Ç–∞—Ö
    """
    height, width = image.shape

    # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —â–æ —Ä–æ–∑–º—ñ—Ä–∏ –∫—Ä–∞—Ç–Ω—ñ 8
    if height % 8 != 0 or width % 8 != 0:
        print(f"‚ö†Ô∏è –£–≤–∞–≥–∞: —Ä–æ–∑–º—ñ—Ä–∏ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è ({height}√ó{width}) –Ω–µ –∫—Ä–∞—Ç–Ω—ñ 8. –ë—É–¥–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–æ {(height//8)*8}√ó{(width//8)*8}")

    Q = get_quantization_table(quality)

    # –ö–æ–Ω–≤–µ—Ä—Ç—É—î–º–æ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –≤ –±—ñ—Ç–∏
    binary_message = text_to_binary(message)
    # –î–æ–¥–∞—î–º–æ –¥–æ–≤–∂–∏–Ω—É –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è (32 –±—ñ—Ç–∏)
    length_binary = format(len(binary_message), '032b')
    binary_data = length_binary + binary_message

    print(f"–ü–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è: {len(message)} —Å–∏–º–≤–æ–ª—ñ–≤")
    print(f"–ë—ñ—Ç–∏ –¥–ª—è –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è: {len(binary_data)} –±—ñ—Ç")

    # DCT –ø–µ—Ä–µ—Ç–≤–æ—Ä–µ–Ω–Ω—è
    dct_image = apply_dct_to_image(image)

    # –ö–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è
    quantized_dct = np.zeros_like(dct_image)
    for i in range(0, height - 7, 8):

    # –í–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è –≤ –∫–≤–∞–Ω—Ç–æ–≤–∞–Ω—ñ –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏
    zigzag_indices = get_zigzag_indices()
    bit_index = 0
    blocks_used = 0

    for block_i in range(0, height - 7, 8):
        for block_j in range(0, width - 7, 8):
            if bit_index >= len(binary_data):
                break

            block = quantized_dct[block_i:block_i+8, block_j:block_j+8].copy()
            block_modified = False

            # –ü—Ä–æ–ø—É—Å–∫–∞—î–º–æ DC –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç, –æ–±—Ä–æ–±–ª—è—î–º–æ —Ç—ñ–ª—å–∫–∏ AC
            for idx in range(1, len(zigzag_indices)):
                if bit_index >= len(binary_data):
                    break

                i, j = zigzag_indices[idx]
                coef = int(block[i, j])

                # JSteg: –ø—Ä–æ–ø—É—Å–∫–∞—î–º–æ 0, 1, -1
                if abs(coef) <= 1:
                    continue

                # –í–±—É–¥–æ–≤—É—î–º–æ –±—ñ—Ç –≤ LSB (–ø—Ä–∞–≤–∏–ª—å–Ω–æ –¥–ª—è –≤—ñ–¥'—î–º–Ω–∏—Ö —á–∏—Å–µ–ª)
                bit = int(binary_data[bit_index])

                sign = 1 if coef >= 0 else -1
                abs_coef = abs(coef)

                # –ó–∞–º—ñ–Ω—é—î–º–æ LSB
                new_abs_coef = (abs_coef & ~1) | bit
                new_coef = sign * new_abs_coef

                block[i, j] = new_coef
                bit_index += 1
                block_modified = True

            if block_modified:
                quantized_dct[block_i:block_i+8, block_j:block_j+8] = block
                blocks_used += 1

        if bit_index >= len(binary_data):
            break

    if bit_index < len(binary_data):
        print(f"‚ö†Ô∏è –ù–µ–¥–æ—Å—Ç–∞—Ç–Ω—å–æ —î–º–Ω–æ—Å—Ç—ñ! –í–±—É–¥–æ–≤–∞–Ω–æ —Ç—ñ–ª—å–∫–∏ {bit_index}/{len(binary_data)} –±—ñ—Ç")
    else:
        print(f"‚úÖ –£—Å–ø—ñ—à–Ω–æ –≤–±—É–¥–æ–≤–∞–Ω–æ {bit_index} –±—ñ—Ç –≤ {blocks_used} –±–ª–æ–∫—ñ–≤")

    # –î–µ–∫–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è
    dequantized_dct = np.zeros_like(dct_image)
    for i in range(0, height - 7, 8):
        for j in range(0, width - 7, 8):
            quant_block = quantized_dct[i:i+8, j:j+8]
            dequantized_dct[i:i+8, j:j+8] = dequantize_dct_block(quant_block, Q)

    # Inverse DCT
    stego_image = apply_idct_to_image(dequantized_dct, image.shape)

    return stego_image, bit_index

def jsteg_extract(stego_image, quality=75):
    """
    –í–∏—Ç—è–≥—É—î –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –º–µ—Ç–æ–¥–æ–º JSteg

    Parameters:
    -----------
    stego_image : numpy array - —Å—Ç–µ–≥–æ-–∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    quality : int - —è–∫—ñ—Å—Ç—å JPEG (–º–∞—î –±—É—Ç–∏ —Ç–∞–∫–∞ –∂, —è–∫ –ø—Ä–∏ –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—ñ)

    dct_image = apply_dct_to_image(stego_image)

    for i in range(0, height - 7, 8):
        for j in range(0, width - 7, 8):
            dct_block = dct_image[i:i+8, j:j+8]
            quantized_dct[i:i+8, j:j+8] = quantize_dct_block(dct_block, Q)

def adaptive_dct_embed(image, message, quality=75, strategy='medium_freq'):
    """
    –í–±—É–¥–æ–≤—É—î –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –∑ –∞–¥–∞–ø—Ç–∏–≤–Ω–∏–º –≤–∏–±–æ—Ä–æ–º –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç—ñ–≤

    Parameters:
    -----------
    image : numpy array - –≤—Ö—ñ–¥–Ω–µ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    message : str - –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è
    quality : int - —è–∫—ñ—Å—Ç—å JPEG
    strategy : str - —Å—Ç—Ä–∞—Ç–µ–≥—ñ—è –≤–∏–±–æ—Ä—É:
        'low_freq' - –Ω–∏–∑—å–∫—ñ —á–∞—Å—Ç–æ—Ç–∏ (1-10 –≤ zigzag)
        'medium_freq' - —Å–µ—Ä–µ–¥–Ω—ñ —á–∞—Å—Ç–æ—Ç–∏ (10-30 –≤ zigzag)
        'high_freq' - –≤–∏—Å–æ–∫—ñ —á–∞—Å—Ç–æ—Ç–∏ (30-64 –≤ zigzag)
        'adaptive' - –Ω–∞ –æ—Å–Ω–æ–≤—ñ —Ç–∞–±–ª–∏—Ü—ñ –∫–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è

    Returns:
    --------
    stego_image : numpy array - —Å—Ç–µ–≥–æ-–∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    capacity : int - –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–∞ —î–º–Ω—ñ—Å—Ç—å
    """
    height, width = image.shape
    Q = get_quantization_table(quality)

    binary_message = text_to_binary(message)
    length_binary = format(len(binary_message), '032b')
    binary_data = length_binary + binary_message

    # DCT —Ç–∞ –∫–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è
    dct_image = apply_dct_to_image(image)
    quantized_dct = np.zeros_like(dct_image)
    for i in range(0, height - 7, 8):
        for j in range(0, width - 7, 8):
            dct_block = dct_image[i:i+8, j:j+8]
            quantized_dct[i:i+8, j:j+8] = quantize_dct_block(dct_block, Q)

    # –í–∏–∑–Ω–∞—á–∞—î–º–æ –¥—ñ–∞–ø–∞–∑–æ–Ω –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç—ñ–≤ –¥–ª—è –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è
    zigzag_indices = get_zigzag_indices()

    if strategy == 'low_freq':
        coef_range = range(1, 10)  # –ù–∏–∑—å–∫—ñ —á–∞—Å—Ç–æ—Ç–∏
    elif strategy == 'medium_freq':
        coef_range = range(10, 30)  # –°–µ—Ä–µ–¥–Ω—ñ —á–∞—Å—Ç–æ—Ç–∏
    elif strategy == 'high_freq':
        coef_range = range(30, 64)  # –í–∏—Å–æ–∫—ñ —á–∞—Å—Ç–æ—Ç–∏
    else:  # adaptive
        coef_range = range(1, 64)

    bit_index = 0

    for block_i in range(0, height - 7, 8):
        for block_j in range(0, width - 7, 8):
            if bit_index >= len(binary_data):
                break

            block = quantized_dct[block_i:block_i+8, block_j:block_j+8].copy()

            for idx in coef_range:
                if bit_index >= len(binary_data):
                    break

                i, j = zigzag_indices[idx]
                coef = int(block[i, j])

                # –ê–¥–∞–ø—Ç–∏–≤–Ω–∞ —Å—Ç—Ä–∞—Ç–µ–≥—ñ—è: –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î–º–æ Q
                if strategy == 'adaptive':
                    # –í–±—É–¥–æ–≤—É—î–º–æ —Ç—ñ–ª—å–∫–∏ —è–∫—â–æ Q < 20 —Ç–∞ |coef| > 3
                    if Q[i, j] > 20 or abs(coef) <= 3:
                        continue
                else:
                    # –î–ª—è —ñ–Ω—à–∏—Ö —Å—Ç—Ä–∞—Ç–µ–≥—ñ–π: –ø—Ä–æ–ø—É—Å–∫–∞—î–º–æ –º–∞–ª—ñ –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏
                    if abs(coef) <= 1:
                        continue

                # –í–±—É–¥–æ–≤—É—î–º–æ –±—ñ—Ç (–ø—Ä–∞–≤–∏–ª—å–Ω–æ –¥–ª—è –≤—ñ–¥'—î–º–Ω–∏—Ö —á–∏—Å–µ–ª)
                bit = int(binary_data[bit_index])

                # –ó–±–µ—Ä—ñ–≥–∞—î–º–æ –∑–Ω–∞–∫ —Ç–∞ –ø—Ä–∞—Ü—é—î–º–æ –∑ –∞–±—Å–æ–ª—é—Ç–Ω–∏–º –∑–Ω–∞—á–µ–Ω–Ω—è–º
                sign = 1 if coef >= 0 else -1
                abs_coef = abs(coef)
                new_abs_coef = (abs_coef & ~1) | bit
                new_coef = sign * new_abs_coef

                block[i, j] = new_coef
                bit_index += 1

            quantized_dct[block_i:block_i+8, block_j:block_j+8] = block

        if bit_index >= len(binary_data):
            break

    # –î–µ–∫–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è —Ç–∞ IDCT
    dequantized_dct = np.zeros_like(dct_image)
    for i in range(0, height - 7, 8):
        for j in range(0, width - 7, 8):
            quant_block = quantized_dct[i:i+8, j:j+8]
            dequantized_dct[i:i+8, j:j+8] = dequantize_dct_block(quant_block, Q)

    stego_image = apply_idct_to_image(dequantized_dct, image.shape)

    return stego_image, bit_index

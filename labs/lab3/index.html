import numpy as np
from PIL import Image
import matplotlib.pyplot as plt

def extract_bit_planes(image_path):
    """–í–∏—Ç—è–≥—É—î –≤—Å—ñ 8 –±—ñ—Ç–æ–≤–∏—Ö –ø–ª–æ—â–∏–Ω —ñ–∑ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è"""
    img = Image.open(image_path).convert('L')  # Grayscale
    img_array = np.array(img)

    bit_planes = []
    for bit in range(8):
        # –í–∏—Ç—è–≥—É—î–º–æ k-–∏–π –±—ñ—Ç
        plane = (img_array >> bit) & 1
        # –ú–∞—Å—à—Ç–∞–±—É—î–º–æ –¥–æ 0-255 –¥–ª—è –≤—ñ–∑—É–∞–ª—ñ–∑–∞—Ü—ñ—ó

        bit_planes.append(plane)

    return bit_planes

from scipy import stats
import numpy as np
from PIL import Image

def chi_square_test(image_path, block_size=128):
    """
    –í–∏–∫–æ–Ω—É—î Chi-square —Ç–µ—Å—Ç –¥–ª—è –≤–∏—è–≤–ª–µ–Ω–Ω—è LSB —Å—Ç–µ–≥–∞–Ω–æ–≥—Ä–∞—Ñ—ñ—ó

    Parameters:
    - image_path: —à–ª—è—Ö –¥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    - block_size: —Ä–æ–∑–º—ñ—Ä –±–ª–æ–∫—É –¥–ª—è –∞–Ω–∞–ª—ñ–∑—É

    Returns:
    - p_values: —Å–ø–∏—Å–æ–∫ p-–∑–Ω–∞—á–µ–Ω—å –¥–ª—è –∫–æ–∂–Ω–æ–≥–æ –±–ª–æ–∫—É
    - detection: True —è–∫—â–æ –≤–∏—è–≤–ª–µ–Ω–æ —Å—Ç–µ–≥–∞–Ω–æ–≥—Ä–∞—Ñ—ñ—é
    """
    img = Image.open(image_path).convert('L')
    img_array = np.array(img)

    height, width = img_array.shape
    p_values = []

    # –ê–Ω–∞–ª—ñ–∑—É—î–º–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è –±–ª–æ–∫–∞–º–∏
    for y in range(0, height - block_size, block_size):
        for x in range(0, width - block_size, block_size):
            block = img_array[y:y+block_size, x:x+block_size].flatten()

            # –ü—ñ–¥—Ä–∞—Ö—É–Ω–æ–∫ —á–∞—Å—Ç–æ—Ç –¥–ª—è –ø–∞—Ä –∑–Ω–∞—á–µ–Ω—å
            hist = np.bincount(block, minlength=256)

            # Chi-square —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
            chi2_stat = 0
            pairs_count = 0

            for i in range(0, 256, 2):
                if i+1 < 256:
                    observed_even = hist[i]
                    observed_odd = hist[i+1]
                    expected = (observed_even + observed_odd) / 2

                    if expected > 0:
                        chi2_stat += ((observed_even - expected)**2 +
                                     (observed_odd - expected)**2) / expected
                        pairs_count += 1

            # –û–±—á–∏—Å–ª—é—î–º–æ p-value
            if pairs_count > 0:
                degrees_of_freedom = pairs_count - 1
                p_value = 1 - stats.chi2.cdf(chi2_stat, degrees_of_freedom)
                p_values.append(p_value)

    # –ê–Ω–∞–ª—ñ–∑ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ–≤
    p_values = np.array(p_values)
    suspicious_blocks = np.sum(p_values < 0.01)
    total_blocks = len(p_values)

    detection_ratio = suspicious_blocks / total_blocks if total_blocks > 0 else 0
    is_stego = detection_ratio > 0.5

    print(f"Chi-square Test Results:")
    print(f"  Total blocks analyzed: {total_blocks}")
    print(f"  Suspicious blocks (p < 0.01): {suspicious_blocks}")
    print(f"  Detection ratio: {detection_ratio:.2%}")
    print(f"  Mean p-value: {np.mean(p_values):.4f}")
    print(f"  Detection: {'STEGO DETECTED!' if is_stego else 'Clean image'}")

    return p_values, is_stego

def plot_chi_square_results(p_values):
    """–í—ñ–∑—É–∞–ª—ñ–∑–∞—Ü—ñ—è —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ–≤ Chi-square —Ç–µ—Å—Ç—É"""
    plt.figure(figsize=(12, 5))

    # –ì—ñ—Å—Ç–æ–≥—Ä–∞–º–∞ p-values
    plt.subplot(1, 2, 1)
    plt.hist(p_values, bins=50, edgecolor='black', alpha=0.7)
    plt.axvline(x=0.01, color='r', linestyle='--', label='Detection threshold')
    plt.xlabel('p-value')
    plt.ylabel('Frequency')
    plt.title('Distribution of Chi-square p-values')
    plt.legend()

    # Heatmap –±–ª–æ–∫—ñ–≤
    plt.subplot(1, 2, 2)
    grid_size = int(np.sqrt(len(p_values)))
    if grid_size**2 == len(p_values):
        p_grid = p_values.reshape(grid_size, grid_size)
        plt.imshow(p_grid, cmap='RdYlGn', vmin=0, vmax=1)
        plt.colorbar(label='p-value')
        plt.title('Spatial distribution of p-values')
    else:
        plt.plot(p_values)
        plt.xlabel('Block index')
        plt.ylabel('p-value')
        plt.title('Sequential p-values')

    plt.tight_layout()
    plt.show()

def rs_analysis(image_path, group_size=4):
    """
    RS (Regular-Singular) –∞–Ω–∞–ª—ñ–∑ –¥–ª—è –≤–∏—è–≤–ª–µ–Ω–Ω—è LSB —Å—Ç–µ–≥–∞–Ω–æ–≥—Ä–∞—Ñ—ñ—ó

    Parameters:
    - image_path: —à–ª—è—Ö –¥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    - group_size: —Ä–æ–∑–º—ñ—Ä –≥—Ä—É–ø–∏ –ø—ñ–∫—Å–µ–ª—ñ–≤ –¥–ª—è –∞–Ω–∞–ª—ñ–∑—É

    Returns:
    - rs_values: —Å–ª–æ–≤–Ω–∏–∫ –∑ RS —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–æ—é
    - is_stego: —Ä–µ–∑—É–ª—å—Ç–∞—Ç –¥–µ—Ç–µ–∫—Ç—É–≤–∞–Ω–Ω—è
    """
    img = Image.open(image_path).convert('L')
    img_array = np.array(img, dtype=np.int32)

    def discrimination_function(group):
        """–û–±—á–∏—Å–ª—é—î –¥–∏—Å–∫—Ä–∏–º—ñ–Ω–∞—Ü—ñ–π–Ω—É —Ñ—É–Ω–∫—Ü—ñ—é"""
        return np.sum(np.abs(np.diff(group)))

    def flip_lsb(value, mask):
        """–§–ªi–ø—É—î LSB –∑–≥—ñ–¥–Ω–æ –∑ –º–∞—Å–∫–æ—é"""
        if mask == 1:
            return value ^ 1  # XOR –∑ 1 —Ñ–ªi–ø—É—î LSB
        return value

    # –ú–∞—Å–∫–∏ –¥–ª—è –∞–Ω–∞–ª—ñ–∑—É
    mask_positive = np.array([1, 0, 1, 0] * (group_size // 4))[:group_size]
    mask_negative = 1 - mask_positive

    # –õ—ñ—á–∏–ª—å–Ω–∏–∫–∏ –¥–ª—è —Ä—ñ–∑–Ω–∏—Ö –∫–∞—Ç–µ–≥–æ—Ä—ñ–π
    R_M, S_M, U_M = 0, 0, 0  # –ü–æ–∑–∏—Ç–∏–≤–Ω–∞ –º–∞—Å–∫–∞
    R_m, S_m, U_m = 0, 0, 0  # –ù–µ–≥–∞—Ç–∏–≤–Ω–∞ –º–∞—Å–∫–∞

    height, width = img_array.shape
    total_groups = 0

    # –ê–Ω–∞–ª—ñ–∑ –≥–æ—Ä–∏–∑–æ–Ω—Ç–∞–ª—å–Ω–∏—Ö –≥—Ä—É–ø
    for y in range(height):
        for x in range(0, width - group_size + 1):
            group = img_array[y, x:x+group_size].copy()

            # –û—Ä–∏–≥—ñ–Ω–∞–ª—å–Ω–∞ –¥–∏—Å–∫—Ä–∏–º—ñ–Ω–∞—Ü—ñ—è
            f_orig = discrimination_function(group)

            # –ó–∞—Å—Ç–æ—Å—É–≤–∞–Ω–Ω—è –ø–æ–∑–∏—Ç–∏–≤–Ω–æ—ó –º–∞—Å–∫–∏
            group_flipped_pos = group.copy()
            for i in range(group_size):
                if mask_positive[i] == 1:
                    group_flipped_pos[i] = flip_lsb(group_flipped_pos[i], 1)
            f_pos = discrimination_function(group_flipped_pos)

            # –ö–ª–∞—Å–∏—Ñ—ñ–∫–∞—Ü—ñ—è –¥–ª—è –ø–æ–∑–∏—Ç–∏–≤–Ω–æ—ó –º–∞—Å–∫–∏
            if f_pos > f_orig:
                R_M += 1
            elif f_pos < f_orig:
                S_M += 1
            else:
                U_M += 1

            # –ó–∞—Å—Ç–æ—Å—É–≤–∞–Ω–Ω—è –Ω–µ–≥–∞—Ç–∏–≤–Ω–æ—ó –º–∞—Å–∫–∏
            group_flipped_neg = group.copy()
            for i in range(group_size):
                if mask_negative[i] == 1:
                    group_flipped_neg[i] = flip_lsb(group_flipped_neg[i], 1)
            f_neg = discrimination_function(group_flipped_neg)

            # –ö–ª–∞—Å–∏—Ñ—ñ–∫–∞—Ü—ñ—è –¥–ª—è –Ω–µ–≥–∞—Ç–∏–≤–Ω–æ—ó –º–∞—Å–∫–∏
            if f_neg > f_orig:
                R_m += 1
            elif f_neg < f_orig:
                S_m += 1
            else:
                U_m += 1

            total_groups += 1

    # –ù–æ—Ä–º–∞–ª—ñ–∑–∞—Ü—ñ—è
    if total_groups > 0:
        R_M = R_M / total_groups
        S_M = S_M / total_groups
        R_m = R_m / total_groups
        S_m = S_m / total_groups

    # RS —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
    x1 = R_M - S_M
    x2 = R_m - S_m

    # –ö—Ä–∏—Ç–µ—Ä—ñ–π –≤–∏—è–≤–ª–µ–Ω–Ω—è
    # –î–ª—è —Å—Ç–µ–≥–æ: |x1| ‚âà |x2|
    # –î–ª—è —á–∏—Å—Ç–æ–≥–æ: |x1| >> |x2|
    ratio = abs(x2) / abs(x1) if abs(x1) > 0.001 else 1

    is_stego = ratio > 0.7  # –ü–æ—Ä—ñ–≥ –¥–ª—è –≤–∏—è–≤–ª–µ–Ω–Ω—è

    rs_values = {
        'R_M': R_M, 'S_M': S_M,
        'R_m': R_m, 'S_m': S_m,
        'x1': x1, 'x2': x2,
        'ratio': ratio
    }

    print(f"RS Analysis Results:")
    print(f"  R_M = {R_M:.4f}, S_M = {S_M:.4f}")
    print(f"  R_-M = {R_m:.4f}, S_-M = {S_m:.4f}")
    print(f"  |R_M - S_M| = {abs(x1):.4f}")
    print(f"  |R_-M - S_-M| = {abs(x2):.4f}")
    print(f"  Ratio = {ratio:.4f}")
    print(f"  Detection: {'STEGO DETECTED!' if is_stego else 'Clean image'}")

    return rs_values, is_stego

class LSBDetector:
    """–ö–æ–º–±—ñ–Ω–æ–≤–∞–Ω–∏–π –¥–µ—Ç–µ–∫—Ç–æ—Ä LSB —Å—Ç–µ–≥–∞–Ω–æ–≥—Ä–∞—Ñ—ñ—ó"""

    def __init__(self):
        self.methods = {
            'chi_square': self.chi_square_detection,
            'rs_analysis': self.rs_detection,
            'histogram': self.histogram_detection,
            'visual': self.visual_detection
        }
        self.weights = {
            'chi_square': 0.3,
            'rs_analysis': 0.3,
            'histogram': 0.2,
            'visual': 0.2
        }

    def chi_square_detection(self, image_path):
        """Chi-square –º–µ—Ç–æ–¥"""
        p_values, is_stego = chi_square_test(image_path, block_size=64)
        confidence = 1 - np.mean(p_values) if len(p_values) > 0 else 0
        return is_stego, confidence

    def rs_detection(self, image_path):
        """RS –∞–Ω–∞–ª—ñ–∑"""
        rs_values, is_stego = rs_analysis(image_path)
        confidence = min(rs_values['ratio'], 1.0) if is_stego else 1 - rs_values['ratio']
        return is_stego, confidence

    def histogram_detection(self, image_path):
        """–ì—ñ—Å—Ç–æ–≥—Ä–∞–º–Ω–∏–π –∞–Ω–∞–ª—ñ–∑"""
        img = Image.open(image_path).convert('L')
        hist, _ = np.histogram(np.array(img), bins=256, range=(0, 256))

        # –ê–Ω–∞–ª—ñ–∑ –ø–∞—Ä –∑–Ω–∞—á–µ–Ω—å
        pair_diff = 0
        for i in range(0, 256, 2):
            if i+1 < 256:
                pair_diff += abs(hist[i] - hist[i+1])

        # –ù–æ—Ä–º–∞–ª—ñ–∑–∞—Ü—ñ—è
        total_pixels = np.sum(hist)
        normalized_diff = pair_diff / total_pixels if total_pixels > 0 else 1

        is_stego = normalized_diff < 0.05
        confidence = 1 - normalized_diff * 10 if is_stego else normalized_diff * 10
        confidence = np.clip(confidence, 0, 1)

        return is_stego, confidence

    def visual_detection(self, image_path):
        """–í—ñ–∑—É–∞–ª—å–Ω–∏–π –∞–Ω–∞–ª—ñ–∑ LSB –ø–ª–æ—â–∏–Ω–∏"""
        img = Image.open(image_path).convert('L')
        img_array = np.array(img)

        # –í–∏—Ç—è–≥—É—î–º–æ LSB –ø–ª–æ—â–∏–Ω—É
        lsb_plane = img_array & 1

        # –û–±—á–∏—Å–ª—é—î–º–æ –µ–Ω—Ç—Ä–æ–ø—ñ—é LSB –ø–ª–æ—â–∏–Ω–∏
        hist, _ = np.histogram(lsb_plane.flatten(), bins=2)
        hist = hist / np.sum(hist)
        entropy = -np.sum(hist * np.log2(hist + 1e-10))

        # –í–∏—Å–æ–∫a –µ–Ω—Ç—Ä–æ–ø—ñ—è –≤–∫–∞–∑—É—î –Ω–∞ –≤–∏–ø–∞–¥–∫–æ–≤—ñ—Å—Ç—å (—Å—Ç–µ–≥–∞–Ω–æ–≥—Ä–∞—Ñ—ñ—è)
        is_stego = entropy > 0.95
        confidence = entropy

        return is_stego, confidence

    def detect(self, image_path, verbose=True):
        """
        –ö–æ–º–±—ñ–Ω–æ–≤–∞–Ω–∏–π –∞–Ω–∞–ª—ñ–∑ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è

        Returns:
        - final_detection: bool
        - confidence: float (0-1)
        - detailed_results: dict
        """
        results = {}
        weighted_score = 0

        for method_name, method_func in self.methods.items():
            try:
                is_detected, confidence = method_func(image_path)
                results[method_name] = {
                    'detected': is_detected,
                    'confidence': confidence
                }

                # –ó–≤–∞–∂–µ–Ω–∞ –æ—Ü—ñ–Ω–∫–∞
                weight = self.weights[method_name]
                weighted_score += confidence * weight if is_detected else (1-confidence) * weight * (-1)

                if verbose:
                    status = "üî¥ DETECTED" if is_detected else "üü¢ CLEAN"
                    print(f"{method_name:15} : {status} (confidence: {confidence:.2%})")

            except Exception as e:
                if verbose:
                    print(f"{method_name:15} : ‚ö†Ô∏è Error - {str(e)}")
                results[method_name] = {'detected': False, 'confidence': 0, 'error': str(e)}

        # –§—ñ–Ω–∞–ª—å–Ω–µ —Ä—ñ—à–µ–Ω–Ω—è
        final_detection = weighted_score > 0
        final_confidence = abs(weighted_score) / sum(self.weights.values())

        if verbose:
            print("\n" + "="*50)
            print(f"FINAL VERDICT: {'üî¥ STEGO DETECTED' if final_detection else 'üü¢ CLEAN IMAGE'}")
            print(f"Confidence: {final_confidence:.2%}")

        return final_detection, final_confidence, results

    def generate_report(self, image_path, output_file="steganalysis_report.html"):
        """–ì–µ–Ω–µ—Ä—É—î HTML –∑–≤—ñ—Ç –ø—Ä–æ –∞–Ω–∞–ª—ñ–∑"""
        detection, confidence, results = self.detect(image_path, verbose=False)

        html_template = """
        
        
        
            Steganalysis Report
            
        
        
            

                
LSB Steganalysis Report

                
Image: {image_path}

                
Analysis Date: {date}

            


            

                
Final Verdict: {verdict}

                
Overall Confidence: {confidence:.2%}

            


            
Detailed Analysis:

            {methods_html}
        
        
        """

        methods_html = ""
        for method, result in results.items():
            status = "Detected" if result['detected'] else "Clean"
            methods_html += f"""
            

                {method}: {status}
                {result['confidence']:.2%}
            

            """

        from datetime import datetime

        html_content = html_template.format(
            image_path=image_path,
            date=datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            verdict="STEGO DETECTED" if detection else "CLEAN IMAGE",
            confidence=confidence,
            result_class="detected" if detection else "clean",
            methods_html=methods_html
        )

        with open(output_file, 'w') as f:
            f.write(html_content)

        print(f"Report saved to {output_file}")

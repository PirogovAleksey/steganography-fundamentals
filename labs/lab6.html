def text_to_binary(text):
    """
    –ö–æ–Ω–≤–µ—Ä—Ç—É—î —Ç–µ–∫—Å—Ç –≤ –±—ñ–Ω–∞—Ä–Ω–∏–π —Ä—è–¥–æ–∫ –∑ –ø—ñ–¥—Ç—Ä–∏–º–∫–æ—é Unicode (UTF-8)

    –í–ê–ñ–õ–ò–í–û: –í–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î UTF-8 –∫–æ–¥—É–≤–∞–Ω–Ω—è –¥–ª—è –ø—ñ–¥—Ç—Ä–∏–º–∫–∏ —É–∫—Ä–∞—ó–Ω—Å—å–∫–æ—ó –º–æ–≤–∏!
    –ö–æ–∂–µ–Ω —Å–∏–º–≤–æ–ª –º–æ–∂–µ –∑–∞–π–º–∞—Ç–∏ 1-4 –±–∞–π—Ç–∏ –≤ –∑–∞–ª–µ–∂–Ω–æ—Å—Ç—ñ –≤—ñ–¥ Unicode –∫–æ–¥—É:
    - ASCII (A-Z, 0-9): 1 –±–∞–π—Ç (8 –±—ñ—Ç)
    - –ö–∏—Ä–∏–ª–∏—Ü—è (–ê-–Ø, –∞-—è): 2 –±–∞–π—Ç–∏ (16 –±—ñ—Ç)
    - Emoji —Ç–∞ —ñ–Ω—à—ñ: 3-4 –±–∞–π—Ç–∏
    """
    utf8_bytes = text.encode('utf-8')
    return ''.join(format(byte, '08b') for byte in utf8_bytes)

def binary_to_text(binary):
    """
    –ö–æ–Ω–≤–µ—Ä—Ç—É—î –±—ñ–Ω–∞—Ä–Ω–∏–π —Ä—è–¥–æ–∫ –≤ —Ç–µ–∫—Å—Ç –∑ –ø—ñ–¥—Ç—Ä–∏–º–∫–æ—é Unicode (UTF-8)

    –í–ê–ñ–õ–ò–í–û: –î–µ–∫–æ–¥—É—î –∑ UTF-8 –¥–ª—è –ø—ñ–¥—Ç—Ä–∏–º–∫–∏ —É–∫—Ä–∞—ó–Ω—Å—å–∫–æ—ó –º–æ–≤–∏!
    """
    # –ö–æ–Ω–≤–µ—Ä—Ç—É—î–º–æ –±—ñ–Ω–∞—Ä–Ω–∏–π —Ä—è–¥–æ–∫ –≤ –±–∞–π—Ç–∏
    byte_array = bytearray()
    for i in range(0, len(binary), 8):
        byte_chunk = binary[i:i+8]
        if len(byte_chunk) == 8:
            byte_array.append(int(byte_chunk, 2))

    # –î–µ–∫–æ–¥—É—î–º–æ –∑ UTF-8
    try:
        return byte_array.decode('utf-8')
    except UnicodeDecodeError:
        return "[–ü–æ–º–∏–ª–∫–∞ –¥–µ–∫–æ–¥—É–≤–∞–Ω–Ω—è UTF-8]"

def calculate_psnr(img1, img2):
    """–û–±—á–∏—Å–ª—é—î PSNR –º—ñ–∂ –¥–≤–æ–º–∞ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è–º–∏"""
    mse = np.mean((img1.astype(float) - img2.astype(float)) ** 2)
    if mse == 0:
        return float('inf')
    max_pixel = 255.0
    psnr = 20 * np.log10(max_pixel / np.sqrt(mse))
    return psnr

def apply_dwt_2d(image, wavelet='haar', level=1):
    """
    –ó–∞—Å—Ç–æ—Å–æ–≤—É—î 2D DWT –¥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è

    Parameters:
    -----------
    image : numpy array - –≤—Ö—ñ–¥–Ω–µ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è (grayscale)
    wavelet : str - —Ç–∏–ø –≤–µ–π–≤–ª–µ—Ç—É ('haar', 'db1', 'db4', 'bior3.3', ...)
    level : int - —Ä—ñ–≤–µ–Ω—å –¥–µ–∫–æ–º–ø–æ–∑–∏—Ü—ñ—ó (1, 2, 3, ...)

    Returns:
    --------
    coeffs : tuple/list - DWT –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏
        –î–ª—è level=1: (LL, (LH, HL, HH))
        –î–ª—è level>1: [LL_n, (LH_n, HL_n, HH_n), ..., (LH_1, HL_1, HH_1)]
    """
    coeffs = pywt.wavedec2(image, wavelet, level=level)
    return coeffs

def apply_idwt_2d(coeffs, wavelet='haar'):
    """
    –ó–∞—Å—Ç–æ—Å–æ–≤—É—î inverse 2D DWT

    Parameters:
    -----------
    coeffs : tuple/list - DWT –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏
    wavelet : str - —Ç–∏–ø –≤–µ–π–≤–ª–µ—Ç—É (–º–∞—î –±—É—Ç–∏ —Ç–æ–π —Å–∞–º–∏–π, —â–æ –ø—Ä–∏ dwt)

    Returns:
    --------
    reconstructed : numpy array - –≤—ñ–¥–Ω–æ–≤–ª–µ–Ω–µ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    """
    reconstructed = pywt.waverec2(coeffs, wavelet)
    return reconstructed

    –ü–æ—Ä—ñ–≤–Ω—é—î —Ä—ñ–∑–Ω—ñ —Ç–∏–ø–∏ –≤–µ–π–≤–ª–µ—Ç—ñ–≤ –Ω–∞ –æ–¥–Ω–æ–º—É –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—ñ

    wavelets = ['haar', 'db2', 'db4', 'sym4', 'bior3.3']

        reconstructed = reconstructed[:test_image.shape[0], :test_image.shape[1]]

        # –ú–µ—Ç—Ä–∏–∫–∏

        LL, (LH, HL, HH) = coeffs
        hh_energy = np.sum(HH ** 2)

        results.append({
            'wavelet': wavelet,
            'psnr': psnr,
            'hh_energy': hh_energy,
            'LL': LL,
            'HH': HH

    plt.show()

    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(12, 5))

    ax2.grid(True, alpha=0.3)

    plt.tight_layout()
    plt.show()

    print("\nüí° –í–∏—Å–Ω–æ–≤–æ–∫:")
    print("  ‚Ä¢ Haar - –Ω–∞–π–ø—Ä–æ—Å—Ç—ñ—à–∏–π, –Ω–∞–π—à–≤–∏–¥—à–∏–π")

def dwt_embed(image, message, wavelet='haar', level=1, subband='HH', delta=10.0):
    """
    –í–±—É–¥–æ–≤—É—î –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –≤ DWT —Å—É–±—Å–º—É–≥—É –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—á–∏ QIM
    (Quantization Index Modulation - –ö–≤–∞–Ω—Ç–æ–≤–∞ –Ü–Ω–¥–µ–∫—Å–Ω–∞ –ú–æ–¥—É–ª—è—Ü—ñ—è)

    Parameters:
    -----------
    image : numpy array - –≤—Ö—ñ–¥–Ω–µ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è (grayscale)
    message : str - –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –¥–ª—è –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è
    wavelet : str - —Ç–∏–ø –≤–µ–π–≤–ª–µ—Ç—É
    level : int - —Ä—ñ–≤–µ–Ω—å –¥–µ–∫–æ–º–ø–æ–∑–∏—Ü—ñ—ó
    subband : str - —Å—É–±—Å–º—É–≥–∞ –¥–ª—è –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è ('LL', 'LH', 'HL', 'HH')
    delta : float - –∫—Ä–æ–∫ –∫–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è (5-15 –¥–ª—è HH, 3-8 –¥–ª—è LH/HL, 1-3 –¥–ª—è LL)

    Returns:
    --------
    stego_image : numpy array - —Å—Ç–µ–≥–æ-–∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    capacity_used : int - –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–∞ —î–º–Ω—ñ—Å—Ç—å –≤ –±—ñ—Ç–∞—Ö
    """
    # –ö–æ–Ω–≤–µ—Ä—Ç—É—î–º–æ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –≤ –±—ñ—Ç–∏
    binary_message = text_to_binary(message)
    # –î–æ–¥–∞—î–º–æ –¥–æ–≤–∂–∏–Ω—É (32 –±—ñ—Ç–∏)
    length_binary = format(len(binary_message), '032b')
    binary_data = length_binary + binary_message

    print(f"–ü–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è: {len(message)} —Å–∏–º–≤–æ–ª—ñ–≤")
    print(f"–ë—ñ—Ç–∏ –¥–ª—è –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è: {len(binary_data)} –±—ñ—Ç")
    print(f"–ü–∞—Ä–∞–º–µ—Ç—Ä–∏: wavelet={wavelet}, level={level}, subband={subband}, Œ¥={delta}")

    # DWT –¥–µ–∫–æ–º–ø–æ–∑–∏—Ü—ñ—è
    coeffs = apply_dwt_2d(image, wavelet=wavelet, level=level)

    # –î–ª—è level=1: (LL, (LH, HL, HH))
    if level == 1:
        LL, (LH, HL, HH) = coeffs

        # –í–∏–±–∏—Ä–∞—î–º–æ —Å—É–±—Å–º—É–≥—É (—Å—Ç–≤–æ—Ä—é—î–º–æ –∫–æ–ø—ñ—é –¥–ª—è –º–æ–¥–∏—Ñ—ñ–∫–∞—Ü—ñ—ó)
        if subband == 'LL':
            target_band = LL.copy()
        elif subband == 'LH':
            target_band = LH.copy()
        elif subband == 'HL':
            target_band = HL.copy()
        else:  # HH
            target_band = HH.copy()
    else:
        # –î–ª—è –±–∞–≥–∞—Ç–æ—Ä—ñ–≤–Ω–µ–≤–æ–≥–æ: –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î–º–æ HH –ø–µ—Ä—à–æ–≥–æ —Ä—ñ–≤–Ω—è
        target_band = coeffs[-1][2].copy()

    # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ —î–º–Ω—ñ—Å—Ç—å
    max_capacity = target_band.size
    print(f"–ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞ —î–º–Ω—ñ—Å—Ç—å —Å—É–±—Å–º—É–≥–∏ {subband}: {max_capacity} –±—ñ—Ç")

    if len(binary_data) > max_capacity:
        print(f"‚ö†Ô∏è –ù–µ–¥–æ—Å—Ç–∞—Ç–Ω—å–æ —î–º–Ω–æ—Å—Ç—ñ! –ü–æ—Ç—Ä—ñ–±–Ω–æ {len(binary_data)}, –¥–æ—Å—Ç—É–ø–Ω–æ {max_capacity}")
        binary_data = binary_data[:max_capacity]

    # QIM –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—è - –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î–º–æ –ø–∞—Ä–Ω—ñ—Å—Ç—å –∫–≤–∞–Ω—Ç–æ–≤–∞–Ω–∏—Ö –∑–Ω–∞—á–µ–Ω—å
    target_band_flat = target_band.flatten()

    for i in range(len(binary_data)):
        bit = int(binary_data[i])
        coeff = target_band_flat[i]

        # –ö–≤–∞–Ω—Ç—É—î–º–æ –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç –¥–æ –Ω–∞–π–±–ª–∏–∂—á–æ–≥–æ –∫—Ä–∞—Ç–Ω–æ–≥–æ delta
        quotient = int(np.round(coeff / delta))

        # –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ –ø–∞—Ä–Ω—ñ—Å—Ç—å —ñ –∫–æ—Ä–∏–≥—É—î–º–æ –ø—Ä–∏ –Ω–µ–æ–±—Ö—ñ–¥–Ω–æ—Å—Ç—ñ
        if bit == 0:
            # –î–ª—è –±—ñ—Ç–∞ 0: –ø–æ—Ç—Ä—ñ–±–Ω–∞ –ü–ê–†–ù–ê –ø–∞—Ä–Ω—ñ—Å—Ç—å (quotient % 2 == 0)
            if quotient % 2 == 1:  # –ó–∞—Ä–∞–∑ –Ω–µ–ø–∞—Ä–Ω–∏–π
                # –ü–µ—Ä–µ–º—ñ—â—É—î–º–æ –¥–æ –Ω–∞–π–±–ª–∏–∂—á–æ–≥–æ –ø–∞—Ä–Ω–æ–≥–æ
                if coeff > quotient * delta:
                    quotient += 1  # –†—É—Ö–∞—î–º–æ—Å—å –≤–≥–æ—Ä—É
                else:
                    quotient -= 1  # –†—É—Ö–∞—î–º–æ—Å—å –≤–Ω–∏–∑
        else:  # bit == 1
            # –î–ª—è –±—ñ—Ç–∞ 1: –ø–æ—Ç—Ä—ñ–±–Ω–∞ –ù–ï–ü–ê–†–ù–ê –ø–∞—Ä–Ω—ñ—Å—Ç—å (quotient % 2 == 1)
            if quotient % 2 == 0:  # –ó–∞—Ä–∞–∑ –ø–∞—Ä–Ω–∏–π
                # –ü–µ—Ä–µ–º—ñ—â—É—î–º–æ –¥–æ –Ω–∞–π–±–ª–∏–∂—á–æ–≥–æ –Ω–µ–ø–∞—Ä–Ω–æ–≥–æ
                if coeff > quotient * delta:
                    quotient += 1
                else:
                    quotient -= 1

        # –ó–∞–ø–∏—Å—É—î–º–æ –∫–≤–∞–Ω—Ç–æ–≤–∞–Ω–µ –∑–Ω–∞—á–µ–Ω–Ω—è
        target_band_flat[i] = quotient * delta

    # –í—ñ–¥–Ω–æ–≤–ª—é—î–º–æ —Ñ–æ—Ä–º—É
    target_band_modified = target_band_flat.reshape(target_band.shape)

    # –û–Ω–æ–≤–ª—é—î–º–æ –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏
    if level == 1:
        if subband == 'LL':
            LL = target_band_modified
        elif subband == 'LH':
            LH = target_band_modified
        elif subband == 'HL':
            HL = target_band_modified
        else:  # HH
            HH = target_band_modified

        coeffs_modified = (LL, (LH, HL, HH))
    else:
        # –î–ª—è –±–∞–≥–∞—Ç–æ—Ä—ñ–≤–Ω–µ–≤–æ–≥–æ
        coeffs[-1] = (coeffs[-1][0], coeffs[-1][1], target_band_modified)
        coeffs_modified = coeffs

    # Inverse DWT
    stego_image = apply_idwt_2d(coeffs_modified, wavelet=wavelet)
    stego_image = stego_image[:image.shape[0], :image.shape[1]]

    # –û–±–º–µ–∂–µ–Ω–Ω—è –¥—ñ–∞–ø–∞–∑–æ–Ω—É [0, 255]
    stego_image = np.clip(stego_image, 0, 255).astype(np.uint8)

    print(f"‚úÖ –í–±—É–¥–æ–≤–∞–Ω–æ {len(binary_data)} –±—ñ—Ç")

    return stego_image, len(binary_data)

def dwt_extract(stego_image, wavelet='haar', level=1, subband='HH', delta=10.0):
    """
    –í–∏—Ç—è–≥—É—î –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è –∑ DWT —Å—É–±—Å–º—É–≥–∏ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—é—á–∏ QIM
    (Quantization Index Modulation - –ö–≤–∞–Ω—Ç–æ–≤–∞ –Ü–Ω–¥–µ–∫—Å–Ω–∞ –ú–æ–¥—É–ª—è—Ü—ñ—è)

    –í–ê–ñ–õ–ò–í–û: –¶–µ –°–õ–Ü–ü–ò–ô (blind) –º–µ—Ç–æ–¥ - –Ω–µ –ø–æ—Ç—Ä–µ–±—É—î –æ—Ä–∏–≥—ñ–Ω–∞–ª—å–Ω–æ–≥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è!

    Parameters:
    -----------
    stego_image : numpy array - —Å—Ç–µ–≥–æ-–∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è
    wavelet : str - —Ç–∏–ø –≤–µ–π–≤–ª–µ—Ç—É (–º–∞—î –±—É—Ç–∏ —Ç–æ–π —Å–∞–º–∏–π, —â–æ –ø—Ä–∏ –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—ñ)
    level : int - —Ä—ñ–≤–µ–Ω—å –¥–µ–∫–æ–º–ø–æ–∑–∏—Ü—ñ—ó
    subband : str - —Å—É–±—Å–º—É–≥–∞
    delta : float - –∫—Ä–æ–∫ –∫–≤–∞–Ω—Ç—É–≤–∞–Ω–Ω—è (–º–∞—î –±—É—Ç–∏ —Ç–æ–π —Å–∞–º–∏–π, —â–æ –ø—Ä–∏ –≤–±—É–¥–æ–≤—É–≤–∞–Ω–Ω—ñ!)

    Returns:
    --------
    extracted_message : str - –≤–∏—Ç—è–≥–Ω—É—Ç–µ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è
    """
    # DWT –¥–µ–∫–æ–º–ø–æ–∑–∏—Ü—ñ—è
    coeffs = apply_dwt_2d(stego_image, wavelet=wavelet, level=level)

    # –í–∏–±–∏—Ä–∞—î–º–æ —Å—É–±—Å–º—É–≥—É
    if level == 1:
        LL, (LH, HL, HH) = coeffs
        if subband == 'LL':
            target_band = LL
        elif subband == 'LH':
            target_band = LH
        elif subband == 'HL':
            target_band = HL
        else:
            target_band = HH
    else:
        target_band = coeffs[-1][2]

    # –í–∏—Ç—è–≥—É–≤–∞–Ω–Ω—è –±—ñ—Ç—ñ–≤ —á–µ—Ä–µ–∑ QIM –¥–µ–∫–æ–¥—É–≤–∞–Ω–Ω—è
    target_band_flat = target_band.flatten()

    # –°–ø–æ—á–∞—Ç–∫—É –≤–∏—Ç—è–≥—É—î–º–æ –¥–æ–≤–∂–∏–Ω—É (32 –±—ñ—Ç–∏)
    extracted_bits = []
    message_length = 0

    for i in range(min(32 + 10000, len(target_band_flat))):  # –û–±–º–µ–∂—É—î–º–æ —Ä–æ–∑—É–º–Ω–æ—é –∫—ñ–ª—å–∫—ñ—Å—Ç—é
        coeff = target_band_flat[i]

        # QIM –¥–µ–∫–æ–¥—É–≤–∞–Ω–Ω—è: –ø–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ –ø–∞—Ä–Ω—ñ—Å—Ç—å –∫–≤–∞–Ω—Ç–æ–≤–∞–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–Ω—è
        quotient = int(np.round(coeff / delta))

        # –ü–∞—Ä–Ω–∏–π quotient ‚Üí –±—ñ—Ç 0, –ù–µ–ø–∞—Ä–Ω–∏–π quotient ‚Üí –±—ñ—Ç 1
        if quotient % 2 == 0:
            bit = 0
        else:
            bit = 1

        extracted_bits.append(str(bit))

        # –ö–æ–ª–∏ –≤–∏—Ç—è–≥–Ω—É–ª–∏ 32 –±—ñ—Ç–∏, –¥—ñ–∑–Ω–∞—î–º–æ—Å—å –¥–æ–≤–∂–∏–Ω—É –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è
        if len(extracted_bits) == 32:
            try:
                message_length = int(''.join(extracted_bits), 2)
                if message_length > 100000 or message_length < 0:
                    return "[–ü–æ–º–∏–ª–∫–∞: –Ω–µ–∫–æ—Ä–µ–∫—Ç–Ω–∞ –¥–æ–≤–∂–∏–Ω–∞ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è]"
            except:
                return "[–ü–æ–º–∏–ª–∫–∞ –¥–µ–∫–æ–¥—É–≤–∞–Ω–Ω—è –¥–æ–≤–∂–∏–Ω–∏]"

        # –ö–æ–ª–∏ –≤–∏—Ç—è–≥–Ω—É–ª–∏ –≤—Å—ñ –±—ñ—Ç–∏ (32 –±—ñ—Ç–∏ –¥–æ–≤–∂–∏–Ω–∏ + –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è)
        if len(extracted_bits) >= 32 + message_length:
            break

    if len(extracted_bits) < 32:
        return "[–ü–æ–º–∏–ª–∫–∞: –Ω–µ–¥–æ—Å—Ç–∞—Ç–Ω—å–æ –¥–∞–Ω–∏—Ö]"

    # –í–∏—Ç—è–≥—É—î–º–æ –±—ñ—Ç–∏ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è (–ø—Ä–æ–ø—É—Å–∫–∞—î–º–æ –ø–µ—Ä—à—ñ 32 –±—ñ—Ç–∏ –¥–æ–≤–∂–∏–Ω–∏)
    message_bits = ''.join(extracted_bits[32:32 + message_length])

    try:
        extracted_message = binary_to_text(message_bits)
        return extracted_message
    except Exception as e:
        return f"[–ü–æ–º–∏–ª–∫–∞ –¥–µ–∫–æ–¥—É–≤–∞–Ω–Ω—è: {str(e)}]"

def apply_attacks(image):
    """
    –ó–∞—Å—Ç–æ—Å–æ–≤—É—î —Ä—ñ–∑–Ω—ñ —Ç–∏–ø–∏ –∞—Ç–∞–∫ –¥–æ –∑–æ–±—Ä–∞–∂–µ–Ω–Ω—è

    """
    attacked = {}

    # 1. JPEG —Å—Ç–∏—Å–Ω–µ–Ω–Ω—è (—Å–∏–º—É–ª—è—Ü—ñ—è)
    encode_param = [int(cv2.IMWRITE_JPEG_QUALITY), 70]
    _, encoded = cv2.imencode('.jpg', image, encode_param)
    attacked['JPEG Q=70'] = cv2.imdecode(encoded, cv2.IMREAD_GRAYSCALE)

    # 2. –ì–∞—É—Å—ñ–≤—Å—å–∫–∏–π —à—É–º
    noise = np.random.normal(0, 5, image.shape)
    attacked['Gaussian Noise œÉ=5'] = np.clip(image.astype(float) + noise, 0, 255).astype(np.uint8)

    # 3. –ú–µ–¥—ñ–∞–Ω–Ω–∏–π —Ñ—ñ–ª—å—Ç—Ä
    attacked['Median Filter 3√ó3'] = cv2.medianBlur(image, 3)

    attacked['Gaussian Blur œÉ=1.0'] = cv2.GaussianBlur(image, (5, 5), 1.0)

    # 5. –ú–∞—Å—à—Ç–∞–±—É–≤–∞–Ω–Ω—è (resize down and up)
    small = cv2.resize(image, (256, 256), interpolation=cv2.INTER_LINEAR)
    attacked['Resize 0.5√ó'] = cv2.resize(small, image.shape[::-1], interpolation=cv2.INTER_LINEAR)

    # 6. –î–æ–¥–∞–≤–∞–Ω–Ω—è —à—É–º—É "—Å—ñ–ª—å-–ø–µ—Ä–µ—Ü—å"
    salt_pepper = image.copy()

    salt_pepper[mask > 1 - prob/2] = 255
    attacked['Salt & Pepper 1%'] = salt_pepper

    return attacked

def calculate_ber(original_bits, extracted_bits):
    """
    –û–±—á–∏—Å–ª—é—î Bit Error Rate –º—ñ–∂ –¥–≤–æ–º–∞ –±—ñ—Ç–æ–≤–∏–º–∏ —Ä—è–¥–∫–∞–º–∏

    Returns:
    --------
    ber : float - BER –≤ –≤—ñ–¥—Å–æ—Ç–∫–∞—Ö
    """
    min_len = min(len(original_bits), len(extracted_bits))
    if min_len == 0:
        return 100.0

    errors = sum(1 for i in range(min_len) if original_bits[i] != extracted_bits[i])
    # –î–æ–¥–∞—î–º–æ –ø–æ–º–∏–ª–∫–∏ –∑–∞ —Ä—ñ–∑–Ω–∏—Ü—é –≤ –¥–æ–≤–∂–∏–Ω—ñ
    errors += abs(len(original_bits) - len(extracted_bits))

    ber = (errors / max(len(original_bits), len(extracted_bits))) * 100
    return ber

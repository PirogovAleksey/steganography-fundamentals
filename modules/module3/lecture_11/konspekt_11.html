def encode_permutation(message_bits, n_packets):
    """–ö–æ–¥—É–≤–∞–Ω–Ω—è —á–∏—Å–ª–∞ —è–∫ –ø–µ—Ä–µ—Å—Ç–∞–Ω–æ–≤–∫–∏ (Lehmer code)"""

        raise ValueError("Message too large for packet count")

    # Lehmer code to permutation
    available = list(range(n_packets))
    permutation = []

    for i in range(n_packets, 0, -1):
        fact = factorial(i - 1)
        index = number // fact
        number = number % fact
        permutation.append(available.pop(index))

    return permutation

def decode_permutation(order):
    """–î–µ–∫–æ–¥—É–≤–∞–Ω–Ω—è –ø–µ—Ä–µ—Å—Ç–∞–Ω–æ–≤–∫–∏ –≤ —á–∏—Å–ª–æ"""

    number = 0
    available = list(range(n))

    for i in range(n):
        index = available.index(order[i])
        number += index * factorial(n - i - 1)
        available.pop(index)

    return number
